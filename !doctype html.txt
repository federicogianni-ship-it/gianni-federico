  <script>
    // ---------------------------
    // Lingua IT/EN (auto-detect + localStorage)
    // ---------------------------
    const LANG_KEY = "site_lang";
    const btnLangIt = document.getElementById("btnLangIt");
    const btnLangEn = document.getElementById("btnLangEn");
    const filmSearch = document.getElementById("filmSearch");

    const i18n = {
      it: {
        title: "Gianni Federico — Cinema e Teatro | Profilo per produzioni",
        placeholderFilmSearch: "Es. Terrarossa, Verdone, 2019..."
      },
      en: {
        title: "Gianni Federico — Film & Theatre | Production profile",
        placeholderFilmSearch: "e.g. Terrarossa, Verdone, 2019..."
      }
    };

    let currentLang = (localStorage.getItem(LANG_KEY) || "").trim();
    if (!currentLang) {
      const n = (navigator.language || "it").toLowerCase();
      currentLang = n.startsWith("en") ? "en" : "it";
    }

    function applyLang(lang) {
      currentLang = lang;
      localStorage.setItem(LANG_KEY, lang);

      // html lang + title
      document.documentElement.lang = lang;
      document.title = i18n[lang].title;

      // mostra/nascondi blocchi
      document.querySelectorAll("[data-lang]").forEach(el => {
        const elLang = el.getAttribute("data-lang");
        if (!elLang) return;
        el.classList.toggle("hidden", elLang !== lang);
      });

      // switch UI
      const activeClasses = ["bg-accent","text-white"];
      const inactiveClasses = ["text-ink","hover:bg-gray-50"];

      [btnLangIt, btnLangEn].forEach(b => {
        b.classList.remove(...activeClasses);
        b.classList.remove(...inactiveClasses);
        b.classList.add(...inactiveClasses);
      });

      const activeBtn = lang === "it" ? btnLangIt : btnLangEn;
      activeBtn.classList.add(...activeClasses);
      activeBtn.classList.remove(...inactiveClasses);

      // placeholder ricerca
      filmSearch.placeholder = i18n[lang].placeholderFilmSearch;

      // reset filtro per evitare "righe nascoste" cambiando lingua
      filmSearch.value = "";
      applyFilmFilter();
    }

    btnLangIt.addEventListener("click", () => applyLang("it"));
    btnLangEn.addEventListener("click", () => applyLang("en"));

    // ---------------------------
    // Smooth scroll + active section
    // ---------------------------
    const navLinks = Array.from(document.querySelectorAll(".navlink"));
    const sections = navLinks
      .map(a => document.querySelector(a.getAttribute("href")))
      .filter(Boolean);

    navLinks.forEach(a => {
      a.addEventListener("click", (e) => {
        const id = a.getAttribute("href");
        const el = document.querySelector(id);
        if (!el) return;
        e.preventDefault();
        el.scrollIntoView({ behavior: "smooth", block: "start" });
        history.pushState(null, "", id);
      });
    });

    const setActive = (id) => {
      navLinks.forEach(a => {
        const isActive = a.getAttribute("href") === id;
        a.classList.toggle("bg-gray-50", isActive);
        a.classList.toggle("font-medium", isActive);
        a.classList.toggle("text-ink", isActive);
      });
    };

    const observer = new IntersectionObserver((entries) => {
      const visible = entries
        .filter(e => e.isIntersecting)
        .sort((a,b) => b.intersectionRatio - a.intersectionRatio)[0];
      if (visible) setActive("#" + visible.target.id);
    }, { rootMargin: "-30% 0px -60% 0px", threshold: [0.1, 0.2, 0.4, 0.6] });

    sections.forEach(s => observer.observe(s));

    // ---------------------------
    // Print
    // ---------------------------
    document.getElementById("btnPrint")?.addEventListener("click", () => window.print());

    // ---------------------------
    // Back to top
    // ---------------------------
    const btnTop = document.getElementById("btnTop");
    const toggleTop = () => {
      const show = window.scrollY > 500;
      btnTop.classList.toggle("hidden", !show);
    };
    window.addEventListener("scroll", toggleTop, { passive:true });
    toggleTop();

    btnTop.addEventListener("click", () => {
      window.scrollTo({ top: 0, behavior: "smooth" });
      history.pushState(null, "", "#");
    });

    // ---------------------------
    // Modal contatti
    // ---------------------------
    const modal = document.getElementById("contactModal");
    const openModal = () => {
      modal.classList.remove("hidden");
      modal.classList.add("flex");
      document.body.classList.add("overflow-hidden");
    };
    const closeModal = () => {
      modal.classList.add("hidden");
      modal.classList.remove("flex");
      document.body.classList.remove("overflow-hidden");
    };

    document.getElementById("btnContact")?.addEventListener("click", openModal);
    document.getElementById("btnCloseModal")?.addEventListener("click", closeModal);
    modal.addEventListener("click", (e) => {
      if (e.target === modal) closeModal();
      if (e.target.classList.contains("bg-black/40")) closeModal();
    });
    window.addEventListener("keydown", (e) => {
      if (e.key === "Escape" && !modal.classList.contains("hidden")) closeModal();
    });

    // ---------------------------
    // Ricerca tabella Cinema (sulla lingua attiva)
    // ---------------------------
    const resetBtn = document.getElementById("filmReset");

    const normalize = (s) => (s || "").toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "");

    function getActiveFilmRows() {
      const tbodyId = currentLang === "it" ? "filmTbodyIt" : "filmTbodyEn";
      return Array.from(document.querySelectorAll(`#${tbodyId} .filmRow`));
    }

    function applyFilmFilter() {
      const q = normalize(filmSearch.value.trim());
      const rows = getActiveFilmRows();
      rows.forEach(r => {
        const txt = normalize(r.innerText);
        const ok = q === "" || txt.includes(q);
        r.classList.toggle("hidden", !ok);
      });
    }

    filmSearch.addEventListener("input", applyFilmFilter);
    resetBtn.addEventListener("click", () => { filmSearch.value = ""; applyFilmFilter(); });

    // Init lingua
    applyLang(currentLang);
